#version 430

in layout(location=0) vec3 position;
in layout(location=1) vec3 inColor;
uniform mat4 transform;
uniform mat4 projection;

out vec3 outColor;

void main()
{
   vec4 v = vec4(position, 1.0);
   vec4 newPos = transform * v;
   newPos = projection * newPos;
   gl_Position = newPos;
   outColor = inColor;
}
